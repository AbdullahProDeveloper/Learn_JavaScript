// ভেরিয়েবল ডিক্লেয়ার করা
var name = "রহিম"; // পুরনো পদ্ধতি
let age = 25; // পরিবর্তনযোগ্য ভেরিয়েবল
const country = "Bangladesh"; // ধ্রুবক ভেরিয়েবল

// ভেরিয়েবল ব্যবহার করা
console.log(name + " এর বয়স " + age + " বছর");
console.log(`${name} lives in ${country}`);

// আউটপুট:
// রহিম এর বয়স ২৫ বছর
// রহিম lives in Bangladesh
// ফাংশন ডিক্লেয়ার করা
function greet(name) {
    return "Hello, " + name + "!";
}

// অ্যারো ফাংশন
const multiply = (a, b) => a * b;

// ফাংশন কল করা
const greeting = greet("আনিকা");
const result = multiply(5, 7);

console.log(greeting); // Hello, আনিকা!
console.log(result); // 35
// for লুপ
for (let i = 1; i <= 5; i++) {
    console.log("Count: " + i);
}

// while লুপ
let count = 1;
while (count <= 5) {
    console.log("While count: " + count);
    count++;
}

// Array এর উপর লুপ
const fruits = ["Apple", "Mango", "Banana"];
for (let fruit of fruits) {
    console.log(fruit);
}

// আউটপুট:
// Count: 1
// Count: 2
// ...
// Apple
// Mango
// Banana

String Reverse
